03 — วงจร (Wiring Diagram)
📌 หมายเหตุ: ESP8266 NodeMCU รองรับแรงดัน 3.3V ที่ขา I/O แต่ขา VIN/5V และ 3V3 สามารถจ่ายไฟให้ MQ-2 ได้ (MQ-2 ใช้ไฟ 5V) — ต่อ VCC ของ MQ-2 เข้ากับขา VIN หรือ 5V ของ NodeMCU
04 — ตารางการต่อสาย
อุปกรณ์ ขาอุปกรณ์ → ขา NodeMCU หมายเหตุ
MQ-2 VCC → VIN / 5V ต้องการ 5V จึงจะทำงาน
MQ-2 AO (Analog) → A0 อ่านค่าความเข้มควัน 0–1023
MQ-2 GND → GND กราวด์ร่วม
LED แดง 🔴 ขายาว (+) → D1 (GPIO5) ผ่านตัวต้านทาน 220Ω
LED แดง 🔴 ขาสั้น (−) → GND —
LED เขียว 🟢 ขายาว (+) → D2 (GPIO4) ผ่านตัวต้านทาน 220Ω
LED เขียว 🟢 ขาสั้น (−) → GND —
Buzzer 🔊 + (บวก) → D5 (GPIO14) Active buzzer ไม่ต้อง PWM
Buzzer 🔊 − (ลบ) → GND —
05 — โค้ด Arduino (ESP8266)
ก่อนใช้งาน ต้องสมัคร LINE Notify ที่ notify-bot.line.me แล้วสร้าง Token จากกลุ่มไลน์ที่ต้องการแจ้งเตือน
// ==================================== // Smoke Detector + LINE Notify // Board: NodeMCU ESP8266 // ==================================== #include <ESP8266WiFi.h> #include <WiFiClientSecure.h> // ---- ตั้งค่าตรงนี้ ---- const char* ssid = "ชื่อ WiFi ของโรงเรียน"; const char* password = "รหัส WiFi"; const char* lineToken = "ใส่ LINE Notify Token ของคุณที่นี่"; const char* location = "ห้องน้ำชั้น 2 ตึก A"; // แก้ตามจุดที่ติดตั้ง // ---- Pin ---- const int PIN_MQ2 = A0; const int PIN_LED_R = D1; const int PIN_LED_G = D2; const int PIN_BUZZ = D5; const int THRESHOLD = 400; // ปรับค่าตามสภาพจริง bool alertSent = false; void setup() { Serial.begin(115200); pinMode(PIN_LED_R, OUTPUT); pinMode(PIN_LED_G, OUTPUT); pinMode(PIN_BUZZ, OUTPUT); // เชื่อมต่อ WiFi WiFi.begin(ssid, password); Serial.print("กำลังเชื่อมต่อ WiFi"); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi Connected!"); digitalWrite(PIN_LED_G, HIGH); // ไฟเขียว = พร้อมทำงาน } void loop() { int smokeVal = analogRead(PIN_MQ2); Serial.print("ค่าควัน: "); Serial.println(smokeVal); if (smokeVal > THRESHOLD) { // พบควัน! digitalWrite(PIN_LED_R, HIGH); digitalWrite(PIN_LED_G, LOW); digitalWrite(PIN_BUZZ, HIGH); if (!alertSent) { String msg = "⚠️ พบควันในห้องน้ำ\n📍 " + String(location) + "\n🔢 ค่าควัน: " + String(smokeVal) + "\n⏰ กรุณาตรวจสอบด่วน!"; sendLINE(msg); alertSent = true; } } else { // ปกติ digitalWrite(PIN_LED_R, LOW); digitalWrite(PIN_LED_G, HIGH); digitalWrite(PIN_BUZZ, LOW); alertSent = false; // reset ให้แจ้งได้ครั้งถัดไป } delay(2000); } void sendLINE(String message) { WiFiClientSecure client; client.setInsecure(); // ข้าม SSL verify (ใช้ในโครงงาน) if (!client.connect("notify-api.line.me", 443)) { Serial.println("เชื่อมต่อ LINE ไม่ได้"); return; } String body = "message=" + urlencode(message); client.println("POST /api/notify HTTP/1.1"); client.println("Host: notify-api.line.me"); client.println("Authorization: Bearer " + String(lineToken)); client.println("Content-Type: application/x-www-form-urlencoded"); client.println("Content-Length: " + String(body.length())); client.println(); client.println(body); Serial.println("ส่งแจ้งเตือน LINE แล้ว ✓"); } String urlencode(String str) { String encoded = ""; for (int i = 0; i < str.length(); i++) { char c = str[i]; if (isAlphaNumeric(c) || c=='-' || c=='_' || c=='.' || c=='~') encoded += c; else { encoded += '%'; encoded += String(c, HEX); } } return encoded; }
06 — ขั้นตอนติดตั้ง Arduino IDE
📦 ติดตั้ง Board & Library
  1. เปิด Arduino IDE → File → Preferences
  2. ใส่ URL: http://arduino.esp8266.com/stable/package_esp8266com_index.json
  3. Tools → Board Manager → ค้นหา esp8266 → Install
  4. เลือก Board: NodeMCU 1.0 (ESP-12E)
  5. เลือก Port: COM port ของ NodeMCU
  6. Upload code → รอ Done
💬 ขอ LINE Notify Token
  1. ไปที่ notify-bot.line.me
  2. Login ด้วยบัญชี LINE
  3. คลิก "Generate token"
  4. ตั้งชื่อ เช่น "แจ้งเตือนห้องน้ำ"
  5. เลือกกลุ่มไลน์ที่ต้องการ
  6. Copy Token ไปใส่ในโค้ด

08 — เคล็ดลับ & การปรับแต่ง
🔧 ปรับค่า Threshold: เปิด Serial Monitor แล้วดูค่า analogRead ในอากาศปกติ จดไว้ แล้วปรับ THRESHOLD ให้สูงกว่าค่าปกติประมาณ 100–200
⏳ Warm-up time: เซ็นเซอร์ MQ-2 ต้องใช้เวลาอุ่นเครื่อง 2–3 นาทีหลังเปิดไฟก่อนค่าจะเสถียร ควรเพิ่ม delay(60000) ใน setup()
📍 ติดตั้งหลายจุด: แต่ละจุดเปลี่ยนตัวแปร location ให้ตรงกับห้อง เช่น "ห้องน้ำชาย ชั้น 1 ตึก B" แล้วใช้ Token เดิม ข้อความจะบอกตำแหน่งชัดเจน
🔋 ไฟเลี้ยง: ในห้องน้ำควรใช้ adapter 5V จาก USB charger ต่อผ่านสาย Micro USB หรือใช้กล่องกันน้ำ IP65 ป้องกันความชื้น
